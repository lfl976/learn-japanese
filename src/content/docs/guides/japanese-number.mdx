## 根据以上阿拉伯数字和日语数字的对应关系，实现一个算法把阿拉伯数字转换为日语数字，注意里面不规则数字的判断

```js
function numberToJapanese(num) {
	const digits = [
		'ゼロ',
		'いち',
		'に',
		'さん',
		'よん',
		'ご',
		'ろく',
		'なな',
		'はち',
		'きゅう',
	];
	const units = ['', 'じゅう', 'ひゃく', 'せん', 'まん', 'おく'];
	const result = [];
	let unitIndex = 0;
	while (num > 0) {
		const digit = num % 10;
		if (digit > 0) {
			result.unshift(units[unitIndex]);
			result.unshift(digits[digit]);
		}
		num = Math.floor(num / 10);
		unitIndex++;
	}
	return result.join('');
}
```
